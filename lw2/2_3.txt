    COACH ||--o{ ACHIEVEMENT : "имеет достижения"
    COACH ||--o{ COACH_CAR_ACCESS : "допуск к авто"
    RACE_CAR ||--o{ COACH_CAR_ACCESS : "доступен тренеру"
    COACH }|--|| RACE_CAR : "любит автомобиль"

    %% Один ко многим
    %% Администратор создает много каналов коммуникации
    %% Канал принадлежит системе (управляется админом)
    ADMIN ||--o{ COMM_CHANNEL : "управляет каналами"

    %% Один ко многим
    %% Клиент приходит через один канал
    %% Канал может привести многих клиентов
    COMM_CHANNEL ||--o{ CLIENT : "привлек клиента"

    %% Рекурсивная связь (Один ко многим)
    %% Лидер (клиент) может привести многих других клиентов
    %% У клиента может быть один лидер (или ни одного)
    CLIENT ||--o{ CLIENT : "пригласил лидера"

    %% Один ко многим
    %% Тренер имеет индивидуальный график работы
    %% График принадлежит одному тренеру
    COACH ||--o{ COACH_SCHEDULE : "имеет график работы"

    %% Один ко многим
    %% Автомобиль может иметь периоды неисправности или обслуживания
    %% Период простоя относится к одному автомобилю
    RACE_CAR ||--o{ CAR_DOWNTIME : "имеет периоды простоя"

    %% Один ко многим
    %% Клуб имеет дни, когда он не работает
    %% Запись о выходном относится к клубу (глобально)
    CLUB_CALENDAR ||--o{ CLUB_HOLIDAY : "имеет выходные дни"

    %% Один ко многим
    %% Тренировка может состоять из нескольких часов (записей времени)
    %% Час тренировки принадлежит одной записи на тренировку
    BOOKING ||--o{ BOOKING_HOUR : "состоит из часов"

    %% Многие ко многим (через сущность BOOKING)
    %% Клиент бронирует тренировку (одна запись - один клиент)
    CLIENT ||--o{ BOOKING : "бронирует тренировку"

    %% Многие ко многим (через сущность BOOKING)
    %% Тренер ведет тренировку (одна запись - один тренер)
    COACH ||--o{ BOOKING : "проводит тренировку"

    %% Многие ко многим (через сущность BOOKING)
    %% Автомобиль используется в тренировке (одна запись - один авто)
    RACE_CAR ||--o{ BOOKING : "используется в тренировке"

    %% Справочник уровней езды (опционально как отдельная сущность или enum)
    %% Здесь реализовано как атрибут в CLIENT, но если нужно редактировать админу:
    %% RIDING_LEVEL ||--o{ CLIENT : "имеет уровень"

    %% === ОПИСАНИЕ ТАБЛИЦ ===

    COMM_CHANNEL {
        int id PK "Уникальный идентификатор"
        string name "Наименование канала"
        date start_date "Дата начала рекламы"
        date end_date "Дата конца (NULL если бессрочно)"
    }

    CLIENT {
        int id PK "Уникальный идентификатор"
        string full_name "ФИО клиента"
        int birth_year "Год рождения"
        string photo_url "Фото (опционально)"
        int channel_id "Ссылка на канал коммуникации"
        int leader_id "Ссылка на лидера-клиента (NULL если нет)"
        string phone_number "Номер телефона"
        string riding_level "Уровень: новичок/средний/профи/мастер"
    }

    COACH_SCHEDULE {
        int id PK "Уникальный идентификатор"
        int coach_id "Ссылка на тренера"
        int day_of_week "День недели (0-6)"
        time start_time "Начало рабочего времени"
        time end_time "Конец рабочего времени"
        date valid_from "Дата начала действия графика"
        date valid_to "Дата окончания действия графика"
    }

    CAR_DOWNTIME {
        int id PK "Уникальный идентификатор"
        int car_id "Ссылка на автомобиль"
        datetime start_datetime "Начало простоя"
        datetime end_datetime "Конец простоя"
        string reason "Причина (ремонт/ТО/другое)"
    }

    CLUB_HOLIDAY {
        int id PK "Уникальный идентификатор"
        date holiday_date "Дата выходного дня"
        string description "Описание (праздник/тех. перерыв)"
    }

    BOOKING {
        int id PK "Уникальный идентификатор"
        int client_id "Ссылка на клиента"
        int coach_id "Ссылка на тренера"
        int car_id "Ссылка на автомобиль"
        string customer_request "Пожелания клиента"
        string status "Статус: забронировано/оплачено/подтверждено/отменено"
        decimal total_cost "Итоговая стоимость"
        datetime created_at "Дата создания брони"
    }

    BOOKING_HOUR {
        int id PK "Уникальный идентификатор"
        int booking_id "Ссылка на бронь"
        date session_date "Дата занятия"
        int hour_start "Час начала (0-23)"
        decimal cost_car "Стоимость аренды авто за час"
        decimal cost_coach "Зарплата тренера за час"
        decimal club_fee "Комиссия клуба за час"
    }
