    %% Один ко многим
    %% Новость содержит много блоков контента
    %% Каждый блок принадлежит ровно одной новости
    NEWS ||--o{ NEWS_BLOCK : "содержит блоки"

    %% Один ко многим (рекурсивная связь)
    %% Комментарий может иметь много ответов (дочерних комментариев)
    %% Ответ принадлежит ровно одному родительскому комментарию
    COMMENT ||--o{ COMMENT : "имеет ответы"

    %% Один ко многим
    %% Новость может иметь много комментариев
    %% Комментарий относится ровно к одной новости
    NEWS ||--o{ COMMENT : "имеет комментарии"

    %% Один ко многим
    %% Новость может иметь много записей о просмотрах
    %% Запись о просмотре относится ровно к одной новости
    NEWS ||--o{ NEWS_VIEW : "имеет просмотры"

    NEWS {
        int id PK "Уникальный идентификатор"
        string title "Заголовок новости"
        boolean is_draft "Статус черновика"
        datetime created_at "Дата создания"
        datetime updated_at "Дата обновления"
    }

    NEWS_BLOCK {
        int id PK "Уникальный идентификатор"
        int news_id "Ссылка на новость"
        int sort_order "Порядковый номер блока"
        string type "Тип блока: image/text"
        text content "Текст или URL изображения"
        boolean is_first_image "Флаг первого изображения (для миниатюры)"
    }

    COMMENT {
        int id PK "Уникальный идентификатор"
        int news_id "Ссылка на новость"
        int parent_comment_id "Ссылка на родительский комментарий (NULL если корневой)"
        text content "Текст комментария"
        string ip_address "IP адрес автора"
        boolean is_deleted "Флаг удаления (мягкое удаление)"
        datetime created_at "Дата создания"
    }

    NEWS_VIEW {
        int id PK "Уникальный идентификатор"
        int news_id "Ссылка на новость"
        string ip_address "IP адрес зрителя"
        datetime viewed_at "Дата и время просмотра"
    }
